   699→// VIEW: ADD / EDIT PRODUCT

   701→function AddProductView() {
   702→  const { navigate, data } = useMerchantNav();
   703→  const isEditing = !!data?.productId;
   704→  const existingProduct = isEditing ? MOCK_PRODUCTS.find((p) => p.id === data.productId) : null;
   705→
   706→  const [categories, setCategories] = useState(CATEGORIES);
   707→  const [categoriesLoaded, setCategoriesLoaded] = useState(false);
   708→
   709→  const [form, setForm] = useState({
   710→    name: existingProduct?.name || '',
   711→    description: existingProduct?.description || '',
   712→    price: existingProduct?.price?.toString() || '',
   713→    comparePrice: existingProduct?.comparePrice?.toString() || '',
   714→    category: existingProduct?.category || '',
   715→    stock: existingProduct?.stock?.toString() || '',
   716→    image: existingProduct?.image || '',
   717→    imageUrl2: '',
   718→    imageUrl3: '',
   719→    variants: '',
   720→  });
   721→  const [saving, setSaving] = useState(false);
   722→
   723→  const update = (key: string, value: string) => setForm((prev) => ({ ...prev, [key]: value }));
   724→
   725→  // Fetch categories from API on first render if not yet loaded
   726→  if (!categoriesLoaded) {
   727→    setCategoriesLoaded(true);
   728→    fetch('/api/categories')
   729→      .then((res) => res.json())
   730→      .then((data) => {
   731→        if (Array.isArray(data) && data.length > 0) {
   732→          setCategories(data.map((c: { name: string; children?: { name: string }[] }) => c.name).flat());
   733→        }
   734→      })
   735→      .catch(() => {});
   736→  }
   737→
   738→  const handleSave = async () => {
   739→    if (!form.name || !form.price || !form.category) {
   740→      toast.error('Veuillez remplir tous les champs obligatoires');
   741→      return;
   742→    }

   745→      const slug = form.name.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '');

   748→        merchantId: 'm1',
   749→        name: form.name,
   750→        description: form.description || null,

   762→        method: 'POST',
   763→        headers: { 'Content-Type': 'application/json' },
   764→        body: JSON.stringify(body),
   765→      });

   772→        comparePrice: '',
   773→        category: '',
   774→        stock: '',
   775→        image: '',
   776→        imageUrl2: '',
   777→        imageUrl3: '',
   778→        variants: '',

   780→      navigate('products');

   785→        description: '',

   787→        comparePrice: '',

   789→        stock: '',
