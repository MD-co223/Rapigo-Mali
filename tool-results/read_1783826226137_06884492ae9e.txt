   699→// VIEW: ADD / EDIT PRODUCT
   700→// ═══════════════════════════════════════════════════════════════════════════════
   701→function AddProductView() {
   702→  const { navigate, data } = useMerchantNav();
   703→  const isEditing = !!data?.productId;
   704→  const existingProduct = isEditing ? MOCK_PRODUCTS.find((p) => p.id === data.productId) : null;
   705→
   706→  const [form, setForm] = useState({
   707→    name: existingProduct?.name || '',
   708→    description: existingProduct?.description || '',
   709→    price: existingProduct?.price?.toString() || '',
   710→    comparePrice: existingProduct?.comparePrice?.toString() || '',
   711→    category: existingProduct?.category || '',
   712→    stock: existingProduct?.stock?.toString() || '',
   713→    image: existingProduct?.image || '',
   714→  });
   715→  const [saving, setSaving] = useState(false);
   716→
   717→  const update = (key: string, value: string) => setForm((prev) => ({ ...prev, [key]: value }));
   718→
   719→  const handleSave = async () => {
   720→    if (!form.name || !form.price || !form.category) {
   721→      toast.error('Veuillez remplir tous les champs obligatoires');
   722→      return;
   723→    }
   724→    setSaving(true);
   725→    try {
   726→      const res = await fetch('/api/products', {
   727→        method: 'POST',
   728→        headers: { 'Content-Type': 'application/json' },
   729→        body: JSON.stringify({
   730→          ...form,
   731→          price: Number(form.price),
   732→          comparePrice: form.comparePrice ? Number(form.comparePrice) : null,
   733→          stock: Number(form.stock) || 0,
   734→          merchantId: 'm1',
   735→        }),
   736→      });
   737→      if (!res.ok) throw new Error('Erreur serveur');
   738→      toast.success(isEditing ? 'Produit mis à jour avec succès' : 'Produit ajouté avec succès');
   739→      navigate('products');
   740→    } catch {
   741→      toast.success(isEditing ? 'Produit mis à jour avec succès' : 'Produit ajouté avec succès');
   742→      navigate('products');
   743→    } finally {
   744→      setSaving(false);
   745→    }
   746→  };
   747→
   748→  return (
   749→    <motion.div {...fadeIn} className="space-y-6 max-w-2xl">
   750→      <div className="flex items-center gap-3">
   751→        <Button variant="ghost" size="icon" onClick={() => navigate('products')}>

   755→          <h1 className="text-2xl font-bold">{isEditing ? 'Modifier le produit' : 'Ajouter un produit'}</h1>
   756→          <p className="text-muted-foreground text-sm mt-0.5">
   757→            {isEditing ? 'Modifiez les informations du produit' : 'Remplissez les informations du nouveau produit'}
   758→          </p>
   759→        </div>
   760→      </div>
   761→
   762→      <Card>
   763→        <CardContent className="p-6 space-y-5">
   764→          {/* Name */}
   765→          <div className="space-y-2">
   766→            <label className="text-sm font-medium">Nom du produit <span className="text-red-500">*</span></label>
   767→            <Input placeholder="Ex: Tô avec sauce arachide" value={form.name} onChange={(e) => update('name', e.target.value)} />
