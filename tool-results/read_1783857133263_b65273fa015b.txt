   156тЖТ   156тЖТfunction useFetch<T>(url: string, fallback: T): { data: T; loading: boolean; refetch: () => void } {
   157тЖТ   157тЖТ  const [data, setData] = useState<T>(fallback);
   158тЖТ   158тЖТ  const [loading, setLoading] = useState(true);
   159тЖТ   159тЖТ
   160тЖТ   160тЖТ  useEffect(() => {
   161тЖТ   161тЖТ    let cancelled = false;
   162тЖТ   162тЖТ    fetch(url)
   163тЖТ   163тЖТ      .then((r) => r.json())
   164тЖТ   164тЖТ      .then((d) => {
   165тЖТ   165тЖТ        if (!cancelled) {
   166тЖТ   166тЖТ          setData(d);
   167тЖТ   167тЖТ          setLoading(false);
   168тЖТ   168тЖТ        }
   169тЖТ   169тЖТ      })
   170тЖТ   170тЖТ      .catch(() => {
   171тЖТ   171тЖТ        if (!cancelled) setLoading(false);
   172тЖТ   172тЖТ      });
   173тЖТ   173тЖТ    return () => { cancelled = true; };
   174тЖТ   174тЖТ  }, [url]);
   175тЖТ   175тЖТ
   176тЖТ   176тЖТ  const refetch = useCallback(() => {
   177тЖТ   177тЖТ    fetch(url)
   178тЖТ   178тЖТ      .then((r) => r.json())
   179тЖТ   179тЖТ      .then((d) => {
   180тЖТ   180тЖТ        setData(d);
   181тЖТ   181тЖТ        setLoading(false);
   182тЖТ   182тЖТ      })
   183тЖТ   183тЖТ      .catch(() => {
   184тЖТ   184тЖТ        setLoading(false);
   185тЖТ   185тЖТ      });
   186тЖТ   186тЖТ  }, [url]);
   187тЖТ   187тЖТ
   188тЖТ   188тЖТ  return { data, loading, refetch };
   189тЖТ   189тЖТ}
