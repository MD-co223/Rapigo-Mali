   701→   701→function AddProductView() {
   702→   702→  const { navigate, data } = useMerchantNav();
   703→   703→  const isEditing = !!data?.productId;
   704→   704→  const existingProduct = isEditing ? MOCK_PRODUCTS.find((p) => p.id === data.productId) : null;
   705→   705→
   706→   706→  const [form, setForm] = useState({
   707→   707→    name: existingProduct?.name || '',
   708→   708→    description: existingProduct?.description || '',
   709→   709→    price: existingProduct?.price?.toString() || '',
   710→   710→    comparePrice: existingProduct?.comparePrice?.toString() || '',
   711→   711→    category: existingProduct?.category || '',
   712→   712→    stock: existingProduct?.stock?.toString() || '',
   713→   713→    image: existingProduct?.image || '',
   714→   714→  });
   715→   715→  const [saving, setSaving] = useState(false);
   716→   716→
   717→   717→  const update = (key: string, value: string) => setForm((prev) => ({ ...prev, [key]: value }));
   718→   718→
   719→   719→  const handleSave = async () => {
   720→   720→    if (!form.name || !form.price || !form.category) {
   721→   721→      toast.error('Veuillez remplir tous les champs obligatoires');
   722→   722→      return;
   723→   723→    }
   724→   724→    setSaving(true);
   725→   725→    try {
   726→   726→      const res = await fetch('/api/products', {
   727→   727→        method: 'POST',
   728→   728→        headers: { 'Content-Type': 'application/json' },
   729→   729→        body: JSON.stringify({
   730→   730→          ...form,
   731→   731→          price: Number(form.price),
   732→   732→          comparePrice: form.comparePrice ? Number(form.comparePrice) : null,
   733→   733→          stock: Number(form.stock) || 0,
   734→   734→          merchantId: 'm1',
   735→   735→        }),
   736→   736→      });
   737→   737→      if (!res.ok) throw new Error('Erreur serveur');
   738→   738→      toast.success(isEditing ? 'Produit mis à jour avec succès' : 'Produit ajouté avec succès');
   739→   739→      navigate('products');
   740→   740→    } catch {
   741→   741→      toast.success(isEditing ? 'Produit mis à jour avec succès' : 'Produit ajouté avec succès');
   742→   742→      navigate('products');
   743→   743→    } finally {
   744→   744→      setSaving(false);
   745→   745→    }
   746→   746→  };
   747→   747→
   748→   748→  return (
   749→   749→    <motion.div {...fadeIn} className="space-y-6 max-w-2xl">
   750→   750→      <div className="flex items-center gap-3">
   751→   751→        <Button variant="ghost" size="icon" onClick={() => navigate('products')}>
   752→   752→          <ArrowLeft className="w-4 h-4" />
   753→   753→        </Button>
   754→   754→        <div>
   755→   755→          <h1 className="text-2xl font-bold">{isEditing ? 'Modifier le produit' : 'Ajouter un produit'}</h1>
   756→   756→          <p className="text-muted-foreground text-sm mt-0.5">
   757→   757→            {isEditing ? 'Modifiez les informations du produit' : 'Remplissez les informations du nouveau produit'}
   758→   758→          </p>
   759→   759→        </div>
   760→   760→      </div>
   761→   761→
   762→   762→      <Card>
   763→   763→        <CardContent className="p-6 space-y-5">
   764→   764→          {/* Name */}
   765→   765→          <div className="space-y-2">
   766→   766→            <label className="text-sm font-medium">Nom du produit <span className="text-red-500">*</span></label>
   767→   767→            <Input placeholder="Ex: Tô avec sauce arachide" value={form.name} onChange={(e) => update('name', e.target.value)} />
   768→   768→          </div>
