   702→/* ── 12. Profile ───────────────────────────── */
   703→function ProfileView() {
   704→  const { user, logout, updateUser } = useAuthStore();
   705→  const { setSpace } = useSpaceStore();
   706→  const [editing, setEditing] = useState(false);
   707→  const [firstName, setFirstName] = useState(user?.firstName || '');
   708→  const [lastName, setLastName] = useState(user?.lastName || '');
   709→  const [phone, setPhone] = useState(user?.phone || '');
   710→  const [saving, setSaving] = useState(false);
   711→
   712→  const handleSave = async () => {
   713→    setSaving(true);
   714→    const res = await apiFetch('/api/auth/me', { method: 'PUT', body: JSON.stringify({ firstName, lastName, phone }) });
   715→    setSaving(false);
   716→    if (res.error) { toast.error(res.error); return; }
   717→    updateUser({ firstName, lastName, phone });
   718→    setEditing(false);
   719→    toast.success('Profil mis à jour');
   720→  };
   721→
   722→  const handleLogout = () => {
   723→    logout();
   724→    setSpace('landing');
   725→  };
   726→
   727→  if (!user) return null;
   728→
   729→  return (
   730→    <div className="px-4 pt-4 space-y-5">
   731→      <h1 className="text-lg font-bold">Mon profil</h1>
   732→      <div className="flex flex-col items-center py-4">
   733→        <Avatar className="h-20 w-20"><AvatarFallback className="bg-emerald-100 text-emerald-700 text-2xl font-bold">{user.firstName.charAt(0)}{user.lastName.charAt(0)}</AvatarFallback></Avatar>
   734→        {!editing && <h2 className="text-xl font-bold mt-3">{user.firstName} {user.lastName}</h2>}
   735→        <p className="text-sm text-muted-foreground">{user.email}</p>
   736→        <p className="text-sm text-muted-foreground">{user.phone}</p>
   737→      </div>
   738→      <Separator />
   739→      {editing ? (
   740→        <div className="space-y-3">
   741→          <div><Label>Prénom</Label><Input value={firstName} onChange={e => setFirstName(e.target.value)} className="mt-1" /></div>
   742→          <div><Label>Nom</Label><Input value={lastName} onChange={e => setLastName(e.target.value)} className="mt-1" /></div>
   743→          <div><Label>Téléphone</Label><Input value={phone} onChange={e => setPhone(e.target.value)} className="mt-1" /></div>
   744→          <div className="flex gap-2"><Button className="flex-1 bg-emerald-700 hover:bg-emerald-800" onClick={handleSave} disabled={saving}>{saving ? <Loader2 className="h-4 w-4 animate-spin mr-1" /> : null}Enregistrer</Button><Button variant="outline" className="flex-1" onClick={() => setEditing(false)}>Annuler</Button></div>
   745→        </div>
   746→      ) : (
   747→        <Button variant="outline" className="w-full" onClick={() => setEditing(true)}>Modifier le profil</Button>
   748→      )}
   749→      <Button variant="outline" className="w-full text-red-600 hover:text-red-700 hover:bg-red-50 border-red-200" onClick={handleLogout}>Déconnexion</Button>
   750→    </div>
   751→  );
   752→}

   754→/* ── 13. Notifications ─────────────────────── */
   755→function NotificationsView() {
   756→  const [notifs, setNotifs] = useState<Notification[]>([]);
   757→  const [loading, setLoading] = useState(true);
   758→
   759→  useEffect(() => { (async () => { const r = await apiFetch<{ notifications: Notification[] }>('/api/notifications'); if (r.data) setNotifs(r.data.notifications); setLoading(false); })(); }, []);
   760→
   761→  const markRead = async (id: string) => {
   762→    await apiFetch(`/api/notifications/${id}/read`, { method: 'PUT' });
   763→    setNotifs(prev => prev.map(n => n.id === id ? { ...n, isRead: true } : n));
   764→  };
   765→
   766→  const markAllRead = async () => {
   767→    await apiFetch('/api/notifications', { method: 'PUT' });
   768→    setNotifs(prev => prev.map(n => ({ ...n, isRead: true })));
   769→    toast.success('Toutes les notifications marquées comme lues');
   770→  };
   771→
   772→  const typeIcon = (type: string) => {
   773→    switch (type) { case 'ORDER': return <Package size={18} />; case 'PAYMENT': return <Wallet size={18} />; case 'PROMO': return <Tag size={18} />; default: return <Bell size={18} />; }
   774→  };
   775→
   776→  return (
   777→    <div className="px-4 pt-4 space-y-4">
   778→      <div className="flex items-center justify-between"><h1 className="text-lg font-bold">Notifications</h1>
   779→        {notifs.some(n => !n.isRead) && <Button variant="ghost" size="sm" className="text-emerald-700 text-xs" onClick={markAllRead}>Tout marquer comme lu</Button>}
   780→      </div>
   781→      {loading ? <Spinner /> : notifs.length === 0 ? <Empty icon={Bell} label="Aucune notification" /> : (
   782→        <div className="space-y-2">{notifs.map(n => (
   783→          <Card key={n.id} className={`shadow-sm cursor-pointer transition-colors ${!n.isRead ? 'bg-emerald-50/50 border-emerald-200' : ''}`} onClick={() => { if (!n.isRead) markRead(n.id); }}>
   784→            <CardContent className="p-3 flex items-start gap-3">
   785→              <div className={`h-9 w-9 rounded-full flex items-center justify-center flex-shrink-0 mt-0.5 ${!n.isRead ? 'bg-emerald-100 text-emerald-700' : 'bg-gray-100 text-gray-500'}`}>{typeIcon(n.type)}</div>
   786→              <div className="flex-1 min-w-0"><p className="text-sm font-medium">{n.title}</p><p className="text-xs text-muted-foreground mt-0.5 line-clamp-2">{n.message}</p><p className="text-[10px] text-muted-foreground mt-1">{new Date(n.createdAt).toLocaleString('fr-FR')}</p></div>
   787→              {!n.isRead && <div className="h-2.5 w-2.5 rounded-full bg-emerald-600 flex-shrink-0 mt-1.5" />}
   788→            </CardContent>
   789→          </Card>
   790→        ))}</div>
   791→      )}
   792→    </div>
   793→  );
   794→}

   827→/* ── 15. Coupons ───────────────────────────── */
   828→function CouponsView() {
   829→  const [code, setCode] = useState('');
   830→  const [result, setResult] = useState<CouponValid | null>(null);
   831→  const [checking, setChecking] = useState(false);
   832→
   833→  const checkCode = async () => {
   834→    if (!code.trim()) return;
   835→    setChecking(true);
   836→    const subtotal = useCartStore.getState().getTotal();
   837→    const res = await apiFetch<CouponValid>('/api/coupons/validate', { method: 'POST', body: JSON.stringify({ code, orderTotal: subtotal, merchantId: useCartStore.getState().merchantId || undefined }) });
   838→    setChecking(false);
   839→    if (res.data) setResult(res.data);
   840→    else setResult({ valid: false, error: res.error || 'Erreur' });
   841→  };
   842→
   843→  return (
   844→    <div className="px-4 pt-4 space-y-5">
   845→      <h1 className="text-lg font-bold">Codes promo</h1>
   846→      <Card className="shadow-sm border-dashed border-2 border-emerald-200 bg-emerald-50/50"><CardContent className="p-6 text-center">
   847→        <Tag className="h-10 w-10 text-emerald-600 mx-auto mb-2" />
   848→        <p className="text-sm font-medium">Vous avez un code promo ?</p>
   849→        <p className="text-xs text-muted-foreground mt-1">Entrez-le ci-dessous pour vérifier sa validité</p>
   850→      </CardContent></Card>
   851→      <div className="flex gap-2">
   852→        <Input value={code} onChange={e => { setCode(e.target.value); setResult(null); }} placeholder="Ex: PROMO2025" className="uppercase" onKeyDown={e => e.key === 'Enter' && checkCode()} />
   853→        <Button className="bg-emerald-700 hover:bg-emerald-800 flex-shrink-0" onClick={checkCode} disabled={checking}>{checking ? <Loader2 className="h-4 w-4 animate-spin" /> : <Check size={16} />}</Button>
   854→      </div>
   855→      {result && (
   856→        <Card className={`shadow-sm ${result.valid ? 'border-emerald-300 bg-emerald-50' : 'border-red-300 bg-red-50'}`}><CardContent className="p-4">
   857→          {result.valid ? (
   858→            <div className="flex items-center gap-2 text-emerald-700"><Check size={18} className="flex-shrink-0" /><div><p className="font-semibold text-sm">Code valide !</p><p className="text-xs mt-0.5">Remise : {formatPrice(result.discount || 0)}</p></div></div>
   859→          ) : (
   860→            <div className="flex items-center gap-2 text-red-600"><X size={18} className="flex-shrink-0" /><p className="text-sm font-medium">{result.error || 'Code invalide'}</p></div>
   861→          )}
   862→        </CardContent></Card>
   863→      )}
   864→    </div>
   865→  );
   866→}
