   179→   179→function formatDate(dateStr: string): string {
   180→   180→  return new Date(dateStr).toLocaleDateString('fr-FR', {
   181→   181→    day: '2-digit',
   182→   182→    month: 'short',
   183→   183→    year: 'numeric',
   184→   184→    hour: '2-digit',
   185→   185→    minute: '2-digit',
   186→   186→  });
   187→   187→}

   189→   189→function formatDateShort(dateStr: string): string {
   190→   190→  return new Date(dateStr).toLocaleDateString('fr-FR', {
   191→   191→    day: '2-digit',
   192→   192→    month: 'short',
   193→   193→    hour: '2-digit',
   194→   194→    minute: '2-digit',

   197→   197→
   198→   198→function timeAgo(dateStr: string): string {
   199→   199→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   200→   200→  if (seconds < 60) return 'À l\'instant';
   201→   201→  if (seconds < 3600) return `Il y a ${Math.floor(seconds / 60)} min`;
   202→   202→  if (seconds < 86400) return `Il y a ${Math.floor(seconds / 3600)} h`;
   203→   203→  return `Il y a ${Math.floor(seconds / 86400)} j`;
   204→   204→}

   365→   365→function HomeView() {

   372→   372→  const { navigate } = useDriverNav();

   374→   374→
   375→   375→  const fetchActiveOrder = useCallback(async () => {
   376→   376→    const res = await apiFetch<Order[]>('/api/orders?limit=50');
   377→   377→    if (res.data) {

   379→   379→        .find((o: Order) => ['ASSIGNED', 'PICKED_UP', 'IN_TRANSIT'].includes(o.status));
   380→   380→      setActiveOrder(active || null);
   381→   381→    }
   382→   382→  }, []);
   383→   383→
   384→   384→  const fetchAvailable = useCallback(async () => {
   385→   385→    const res = await apiFetch<Order[]>('/api/drivers/available-orders');
   386→   386→    if (res.data) {
   387→   387→      setAvailableOrders(Array.isArray(res.data) ? res.data : []);
   388→   388→    }
   389→   389→  }, []);
   390→   390→
   391→   391→  const fetchStats = useCallback(async () => {
   392→   392→    const res = await apiFetch<Order[]>('/api/orders?limit=200');
   393→   393→    if (res.data) {
   394→   394→      const orders = Array.isArray(res.data) ? res.data : (res.data as unknown as { orders: Order[] }).orders || [];
   395→   395→      const today = new Date();
   396→   396→      today.setHours(0, 0, 0, 0);
   397→   397→      const todayOrders = orders.filter(
   398→   398→        (o: Order) => o.status === 'DELIVERED' && new Date(o.deliveredAt || o.createdAt) >= today
   399→   399→      );
   400→   400→      const deliveredOrders = orders.filter((o: Order) => o.status === 'DELIVERED');
   401→   401→      const totalRevenue = deliveredOrders.reduce((sum: number, o: Order) => sum + o.deliveryFee, 0);
   402→   402→      const ratings = orders
   403→   403→        .filter((o: Order) => o.ratings && o.ratings.length > 0)
   404→   404→        .flatMap((o: Order) => o.ratings || [])
   405→   405→        .map((r: { rating: number }) => r.rating);
   406→   406→      const avgRating = ratings.length > 0 ? ratings.reduce((a: number, b: number) => a + b, 0) / ratings.length : 0;
   407→   407→      setStats({ today: todayOrders.length, revenue: totalRevenue, avgRating });
   408→   408→    }
   409→   409→  }, []);

   412→   412→    let mounted = true;

   414→   414→      const [ordersRes, availableRes, statsRes] = await Promise.all([
   415→   415→        apiFetch<Order[]>('/api/orders?limit=50'),
   416→   416→        apiFetch<Order[]>('/api/drivers/available-orders'),
   417→   417→        apiFetch<Order[]>('/api/orders?limit=200'),
   418→   418→      ]);

   421→   421→        const list = Array.isArray(ordersRes.data) ? ordersRes.data : (ordersRes.data as unknown as { orders: Order[] }).orders || [];
   422→   422→        setActiveOrder(list.find((o: Order) => ['ASSIGNED', 'PICKED_UP', 'IN_TRANSIT'].includes(o.status)) || null);
   423→   423→      }
   424→   424→      if (availableRes.data) {
   425→   425→        setAvailableOrders(Array.isArray(availableRes.data) ? availableRes.data : []);
   426→   426→      }
   427→   427→      if (statsRes.data) {
   428→   428→        const orders = Array.isArray(statsRes.data) ? statsRes.data : (statsRes.data as unknown as { orders: Order[] }).orders || [];
   429→   429→        const today = new Date();
   430→   430→        today.setHours(0, 0, 0, 0);
   431→   431→        const todayOrders = orders.filter(
   432→   432→          (o: Order) => o.status === 'DELIVERED' && new Date(o.deliveredAt || o.createdAt) >= today
   433→   433→        );
   434→   434→        const deliveredOrders = orders.filter((o: Order) => o.status === 'DELIVERED');
   435→   435→        const totalRevenue = deliveredOrders.reduce((sum: number, o: Order) => sum + o.deliveryFee, 0);
   436→   436→        const ratings = orders
   437→   437→          .filter((o: Order) => o.ratings && o.ratings.length > 0)
   438→   438→          .flatMap((o: Order) => o.ratings || [])
   439→   439→          .map((r: { rating: number }) => r.rating);
   440→   440→        const avgRating = ratings.length > 0 ? ratings.reduce((a: number, b: number) => a + b, 0) / ratings.length : 0;
   441→   441→        setStats({ today: todayOrders.length, revenue: totalRevenue, avgRating });
   442→   442→      }
   443→   443→      setLoading(false);
   444→   444→    };
   445→   445→    init();
   446→   446→    return () => { mounted = false; };
   447→   447→  }, []);
   448→   448→
   449→   449→  useEffect(() => {
   450→   450→    if (isOnline && !activeOrder) {
   451→   451→      timerRef.current = setInterval(() => {
   452→   452→        fetchAvailable();
   453→   453→        fetchActiveOrder();
   454→   454→      }, 15000);
   455→   455→    }
   456→   456→    return () => {
   457→   457→      if (timerRef.current) clearInterval(timerRef.current);
   458→   458→    };
   459→   459→  }, [isOnline, activeOrder, fetchAvailable, fetchActiveOrder]);
   460→   460→
   461→   461→  const handleToggle = () => {
   462→   462→    const next = !isOnline;
   463→   463→    setIsOnline(next);
   464→   464→    toast.success(next ? 'Vous êtes en ligne' : 'Vous êtes hors ligne');
   465→   465→    if (next) {
