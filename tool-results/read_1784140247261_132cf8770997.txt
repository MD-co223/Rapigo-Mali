   156→   156→function useFetch<T>(url: string, fallback: T): { data: T; loading: boolean; refetch: () => void } {
   157→   157→  const [data, setData] = useState<T>(fallback);
   158→   158→  const [loading, setLoading] = useState(true);
   159→   159→
   160→   160→  useEffect(() => {
   161→   161→    let cancelled = false;
   162→   162→    fetch(url)
   163→   163→      .then((r) => r.json())
   164→   164→      .then((d) => {
   165→   165→        if (!cancelled) {
   166→   166→          setData(d);
   167→   167→          setLoading(false);
   168→   168→        }
   169→   169→      })
   170→   170→      .catch(() => {
   171→   171→        if (!cancelled) setLoading(false);
   172→   172→      });
   173→   173→    return () => { cancelled = true; };
   174→   174→  }, [url]);
   175→   175→
   176→   176→  const refetch = useCallback(() => {
   177→   177→    fetch(url)
   178→   178→      .then((r) => r.json())
   179→   179→      .then((d) => {
   180→   180→        setData(d);
   181→   181→        setLoading(false);
   182→   182→      })
   183→   183→      .catch(() => {
   184→   184→        setLoading(false);
   185→   185→      });
   186→   186→  }, [url]);
   187→   187→
   188→   188→  return { data, loading, refetch };
   189→   189→}
