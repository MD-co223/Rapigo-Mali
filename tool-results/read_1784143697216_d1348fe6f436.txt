   211→function useApiFetch<T>(url: string, fallback: T) {
   212→  const [data, setData] = useState<T>(fallback);
   213→  const [loading, setLoading] = useState(true);
   214→  const [error, setError] = useState<string | null>(null);
   215→
   216→  const fetch_ = useCallback(async () => {
   217→    setLoading(true);
   218→    setError(null);
   219→    const res = await apiFetch<T>(url);
   220→    if (res.error) {
   221→      setError(res.error);
   222→    } else if (res.data) {
   223→      setData(res.data);
   224→    }
   225→    setLoading(false);
   226→  }, [url]);
   227→
   228→  useEffect(() => { fetch_(); }, [fetch_]);
   229→
   230→  return { data, loading, error, refetch: fetch_ };
   231→}
